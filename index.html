function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('CallOuts'); if (!sheet) return respond({ error: 'Sheet not found' }); if (data.action === 'add') { sheet.appendRow([data.id, data.date, data.name, data.reason]); return respond({ success: true }); } if (data.action === 'delete') { const rows = sheet.getDataRange().getValues(); for (let i = rows.length - 1; i >= 1; i--) { if (String(rows[i][0]) === String(data.id)) { sheet.deleteRow(i + 1); return respond({ success: true }); } } return respond({ error: 'Row not found' }); } return respond({ error: 'Unknown action' }); } catch(err) { return respond({ error: err.toString() }); } } function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('CallOuts'); if (!sheet) return respond({ error: 'Sheet not found' }); const rows = sheet.getDataRange().getValues(); const entries = rows.slice(1).map(r => ({ id: r[0], date: r[1], name: r[2], reason: r[3] })); return respond({ entries }); } catch(err) { return respond({ error: err.toString() }); } } function respond(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
Instructions
1
Create a new Google Sheet using your personal Gmail account (work accounts may restrict script deployments).
2
Rename the first tab to exactly CallOuts. Then manually add headers: ID in A1, Date in B1, Name in C1, Reason in D1.
3
Go to Extensions → Apps Script. Delete any existing code. and paste it into the editor, then click Save.
4
Click Deploy → New Deployment. Set type to Web App, execute as Me, access to Anyone. Copy the deployment URL.
5
Paste your deployment URL below and click Connect. The status dot will turn green when connected.
6
You may notice a brief delay when adding or deleting entries - this is normal. Each change routes through Google's servers, which adds a second or two compared to local storage.
